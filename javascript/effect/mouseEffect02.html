    const cursor = document.querySelector(".cursor")
    const follower = document.querySelector(".cursor-follower")
    const span = document.querySelectorAll(".mouse__wrap span")
    const info = document.querySelectorAll(".number ul li")
    const title = document.querySelectorAll(".title")

    window.addEventListener("mousemove", e => {
        //커서 좌표값 할당
        // cursor.style.left = e.pageX -5 + "px"
        // cursor.style.top = e.pageY-5 + "px"
        // follower.style.left = e.pageX -15 + "px"
        // follower.style.top = e.pageY - 15 + "px"

        //gsap
        gsap.to(cursor, {
            duration: .3,
            left: e.pageX - 5,
            top: e.pageY - 5
        })
        gsap.to(follower, {
            duration: .8,
            left: e.pageX - 15,
            top: e.pageY - 15
        })

        //오버 효과
        //mouseover, mouseenter / mouseout,mouseleave / 이벤트 버블링 차이 정리.
        span.forEach((e) => {
            e.addEventListener("mouseover", () => {
                cursor.classList.add("active")
                follower.classList.add("active")
            })
            e.addEventListener("mouseout", () => {
                cursor.classList.remove("active")
                follower.classList.remove("active")
            })
        })
        //inof
        info.forEach((e) => {
            e.addEventListener("mouseover", () => {
                cursor.classList.add("show")
                follower.classList.add("show")
            })
            e.addEventListener("mouseout", () => {
                cursor.classList.remove("show")
                follower.classList.remove("show")
            })
        })
        //title
        title.forEach((e) => {
            e.addEventListener("mouseover", () => {
                cursor.classList.add("show2")
                follower.classList.add("show2")
            })
            e.addEventListener("mouseout", () => {
                cursor.classList.remove("show2")
                follower.classList.remove("show2")
            })
        })
        //span 오버 했을 때 클래스 active 추가 / 나갔을 때 active 삭제

        //출력
        document.querySelector(".pageX").innerText = e.pageX
        document.querySelector(".pageY").innerText = e.pageY
    })
