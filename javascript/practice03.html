1.1 테이블 출력하기

{
    let table = "<table border='1'>";

        for( let i = 1; i<=5; i++){
            table +="<tr>";

                for( let j=1; j<=5; j++){
                    table += "<td>1</td>";
                }
            table +="</tr>";
        }

        table += "</table>";
        document.write(table)
}
